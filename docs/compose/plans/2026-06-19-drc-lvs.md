```rust
pub struct DrcChecker {
    rules: DrcRuleSet,
}

#[derive(Debug, Clone, PartialEq)]
pub struct DrcRuleSet {
    pub min_trace_spacing_um: f64,
    pub min_jj_spacing_um: f64,
    pub max_ptl_length_um: Option<f64>,
    pub cell_boundary_margin_um: f64,
    pub layout_width_um: f64,
    pub layout_height_um: f64,
}

impl DrcRuleSet {
    pub fn from_pdk(pdk: &Pdk, layout_width: f64, layout_height: f64) -> Self {
        let (spacing, jj_spacing, margin) = if let Some(ref rules) = pdk.drc_rules {
            (rules.min_trace_spacing_um, rules.min_jj_spacing_um, rules.cell_boundary_margin_um)
        } else {
            (1.0, 5.0, 2.0)
        };
        let max_ptl = pdk.ptl_forbidden_ranges.iter()
            .map(|r| r.max_um)
            .fold(0.0f64, f64::max);
        let max_ptl = if max_ptl > 0.0 { Some(max_ptl) } else { None };
        Self {
            min_trace_spacing_um: spacing,
            min_jj_spacing_um: jj_spacing,
            max_ptl_length_um: max_ptl,
            cell_boundary_margin_um: margin,
            layout_width_um: layout_width,
            layout_height_um: layout_height,
        }
    }
}

impl DrcChecker {
    pub fn new(rules: DrcRuleSet) -> Self {
        Self { rules }
    }

    pub fn check(
        &self,
        placement: &Placement,
        routing: &RoutingReport,
        netlist: &Netlist,
    ) -> DrcReport {
        let mut violations = Vec::new();
        let mut checked = 0usize;

        // 1. Trace spacing
        checked += 1;
        violations.extend(self.check_trace_spacing(&routing.routes));

        // 2. PTL length
        checked += 1;
        violations.extend(self.check_ptl_length(&routing.routes));

        // 3. JJ spacing
        checked += 1;
        violations.extend(self.check_jj_spacing(placement, netlist));

        // 4. Cell boundary
        checked += 1;
        violations.extend(self.check_cell_boundary(placement));

        let error_count = violations.iter()
            .filter(|v| v.severity == DrcSeverity::Error)
            .count();
        let warning_count = violations.iter()
            .filter(|v| v.severity == DrcSeverity::Warning)
            .count();

        DrcReport {
            violations,
            checked_rules: checked,
            error_count,
            warning_count,
            passed: error_count == 0,
        }
    }

    fn check_trace_spacing(&self, routes: &[rflux_route::NetRoute]) -> Vec<DrcViolation> {
        let mut violations = Vec::new();
        // Check all pairs of segments from different nets
        for (i, r1) in routes.iter().enumerate() {
            for s1 in &r1.segments {
                for (j, r2) in routes.iter().enumerate() {
                    if i >= j { continue; } // avoid duplicate pairs
                    if r1.from == r2.from && r1.to == r2.to { continue; } // same net
                    for s2 in &r2.segments {
                        if s1.layer != s2.layer { continue; } // different layers OK
                        let dist = segment_distance(s1, s2);
                        if dist < self.rules.min_trace_spacing_um {
                            violations.push(DrcViolation {
                                rule: "trace_spacing".to_string(),
                                severity: DrcSeverity::Error,
                                location: Some(midpoint(s1)),
                                detail: format!(
                                    "nets {:?}-{:?} and {:?}-{:?} on layer {} are {:.2}um apart (min {:.2}um)",
                                    r1.from, r1.to, r2.from, r2.to, s1.layer, dist, self.rules.min_trace_spacing_um
                                ),
                            });
                        }
                    }
                }
            }
        }
        violations
    }

    fn check_ptl_length(&self, routes: &[rflux_route::NetRoute]) -> Vec<DrcViolation> {
        let mut violations = Vec::new();
        if let Some(max_length) = self.rules.max_ptl_length_um {
            for route in routes {
                if route.mode == rflux_route::RouteMode::Ptl && route.length_um > max_length {
                    violations.push(DrcViolation {
                        rule: "ptl_length".to_string(),
                        severity: DrcSeverity::Warning,
                        location: route.segments.first().map(|s| s.start),
                        detail: format!(
                            "PTL route {:?}-{:?} is {:.2}um (max {:.2}um)",
                            route.from, route.to, route.length_um, max_length
                        ),
                    });
                }
            }
        }
        violations
    }

    fn check_jj_spacing(&self, placement: &Placement, netlist: &Netlist) -> Vec<DrcViolation> {
        let mut violations = Vec::new();
        let cells: Vec<_> = placement.nodes.iter()
            .filter(|p| {
                netlist.nodes().get(p.node.0)
                    .map(|n| matches!(n.kind, rflux_ir::NodeKind::CellInstance | rflux_ir::NodeKind::Dff | rflux_ir::NodeKind::MacroCell))
                    .unwrap_or(false)
            })
            .collect();

        for (i, c1) in cells.iter().enumerate() {
            for c2 in cells.iter().skip(i + 1) {
                let dx = c1.point.x_um - c2.point.x_um;
                let dy = c1.point.y_um - c2.point.y_um;
                let dist = (dx * dx + dy * dy).sqrt();
                if dist < self.rules.min_jj_spacing_um {
                    let n1 = netlist.nodes().get(c1.node.0).map(|n| n.name.clone()).unwrap_or_default();
                    let n2 = netlist.nodes().get(c2.node.0).map(|n| n.name.clone()).unwrap_or_default();
                    violations.push(DrcViolation {
                        rule: "jj_spacing".to_string(),
                        severity: DrcSeverity::Error,
                        location: Some(Point {
                            x_um: (c1.point.x_um + c2.point.x_um) / 2.0,
                            y_um: (c1.point.y_um + c2.point.y_um) / 2.0,
                        }),
                        detail: format!(
                            "cells '{}' and '{}' are {:.2}um apart (min {:.2}um)",
                            n1, n2, dist, self.rules.min_jj_spacing_um
                        ),
                    });
                }
            }
        }
        violations
    }

    fn check_cell_boundary(&self, placement: &Placement) -> Vec<DrcViolation> {
        let mut violations = Vec::new();
        let m = self.rules.cell_boundary_margin_um;
        for placed in &placement.nodes {
            if placed.point.x_um < m || placed.point.x_um > self.rules.layout_width_um - m
                || placed.point.y_um < m || placed.point.y_um > self.rules.layout_height_um - m
            {
                violations.push(DrcViolation {
                    rule: "cell_boundary".to_string(),
                    severity: DrcSeverity::Warning,
                    location: Some(placed.point),
                    detail: format!(
                        "node {:?} at ({:.2}, {:.2}) is within {:.2}um of layout boundary",
                        placed.node, placed.point.x_um, placed.point.y_um, m
                    ),
                });
            }
        }
        violations
    }
}

fn segment_distance(s1: &RouteSegment, s2: &RouteSegment) -> f64 {
    // Manhattan distance between closest points on two segments
    let points1 = [(s1.start.x_um, s1.start.y_um), (s1.end.x_um, s1.end.y_um)];
    let points2 = [(s2.start.x_um, s2.start.y_um), (s2.end.x_um, s2.end.y_um)];
    let mut min_dist = f64::INFINITY;
    for &(x1, y1) in &points1 {
        for &(x2, y2) in &points2 {
            let d = (x1 - x2).abs() + (y1 - y2).abs();
            min_dist = min_dist.min(d);
        }
    }
    min_dist
}

fn midpoint(s: &RouteSegment) -> Point {
    Point {
        x_um: (s.start.x_um + s.end.x_um) / 2.0,
        y_um: (s.start.y_um + s.end.y_um) / 2.0,
    }
}
```

```rust
pub struct LvsChecker;

impl LvsChecker {
    pub fn new() -> Self {
        Self
    }

    pub fn check(
        &self,
        netlist: &Netlist,
        placement: &Placement,
        routing: &RoutingReport,
    ) -> LvsReport {
        // 1. Count devices in schematic
        let schematic_devices: Vec<_> = netlist.nodes()
            .filter(|n| matches!(n.kind,
                rflux_ir::NodeKind::CellInstance | rflux_ir::NodeKind::Dff |
                rflux_ir::NodeKind::MacroCell | rflux_ir::NodeKind::Splitter
            ))
            .collect();

        // 2. Count devices in layout
        let layout_devices: Vec<_> = placement.nodes.iter()
            .filter(|p| {
                netlist.nodes().get(p.node.0)
                    .map(|n| matches!(n.kind,
                        rflux_ir::NodeKind::CellInstance | rflux_ir::NodeKind::Dff |
                        rflux_ir::NodeKind::MacroCell | rflux_ir::NodeKind::Splitter
                    ))
                    .unwrap_or(false)
            })
            .collect();

        let device_count_mismatch = schematic_devices.len() != layout_devices.len();

        // 3. Check connectivity
        let schematic_nets = extract_nets_from_schematic(netlist);
        let layout_nets = extract_nets_from_layout(routing);
        let mut net_mismatches = Vec::new();
        let mut matched_nets = 0usize;

        for (name, schematic_pins) in &schematic_nets {
            if let Some(layout_pins) = layout_nets.get(name) {
                if schematic_pins == layout_pins {
                    matched_nets += 1;
                } else {
                    net_mismatches.push(NetMismatch {
                        net_name: name.clone(),
                        issue: format!("connectivity mismatch: schematic {:?} vs layout {:?}", schematic_pins, layout_pins),
                    });
                }
            } else {
                net_mismatches.push(NetMismatch {
                    net_name: name.clone(),
                    issue: "net exists in schematic but not in layout".to_string(),
                });
            }
        }

        let connectivity_mismatch = !net_mismatches.is_empty();
        let checked_nets = schematic_nets.len();

        LvsReport {
            matched: !device_count_mismatch && !connectivity_mismatch,
            device_count_mismatch,
            connectivity_mismatch,
            missing_devices: Vec::new(), // TODO: compute diff
            extra_devices: Vec::new(),
            net_mismatches,
            checked_nets,
            matched_nets,
        }
    }
}
```

```rust
use std::collections::{HashMap, BTreeSet};

fn extract_nets_from_schematic(netlist: &Netlist) -> HashMap<String, BTreeSet<usize>> {
    let mut nets: HashMap<String, BTreeSet<usize>> = HashMap::new();
    for edge in netlist.edges() {
        let from_node = &netlist.nodes()[edge.from.node.0];
        let to_node = &netlist.nodes()[edge.to.node.0];
        let net_name = format!("{}_to_{}", from_node.name, to_node.name);
        nets.entry(net_name).or_default().insert(edge.from.node.0);
        nets.entry(net_name).or_default().insert(edge.to.node.0);
    }
    nets
}

fn extract_nets_from_layout(routing: &RoutingReport) -> HashMap<String, BTreeSet<usize>> {
    let mut nets: HashMap<String, BTreeSet<usize>> = HashMap::new();
    for route in &routing.routes {
        let net_name = format!("{:?}_to_{:?}", route.from.node, route.to.node);
        nets.entry(net_name).or_default().insert(route.from.node.0);
        nets.entry(net_name).or_default().insert(route.to.node.0);
    }
    nets
}
```

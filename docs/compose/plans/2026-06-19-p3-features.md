- [ ] **Step 2: 实现 parse_spice_netlist**

```rust
pub fn parse_spice_netlist(deck: &str) -> Result<Vec<SpiceDevice>, SimulationError> {
    let lines = preprocess_deck(deck);
    let mut devices = Vec::new();
    for line in &lines {
        let trimmed = line.trim();
        if trimmed.is_empty() || trimmed.starts_with('.') {
            continue;
        }
        if let Some(device) = parse_spice_line(trimmed)? {
            devices.push(device);
        }
    }
    Ok(devices)
}

fn parse_spice_line(line: &str) -> Result<Option<SpiceDevice>, SimulationError> {
    let first_char = line.chars().next().unwrap_or(' ');
    match first_char {
        'X' | 'x' => {
            // Subcircuit instance: Xname net1 net2 ... subckt_name
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() < 3 { return Err(SimulationError::InvalidSubcktInstance(line.to_string())); }
            let name = parts[0].to_string();
            let subckt_name = parts.last().unwrap().to_string();
            let connections = parts[1..parts.len()-1].iter().map(|s| s.to_string()).collect();
            Ok(Some(SpiceDevice { name, kind: SpiceDeviceKind::SubcktInstance(subckt_name), connections, params: BTreeMap::new() }))
        }
        'J' | 'j' => {
            // JJ: Jname n+ n- model [params]
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() < 4 { return Err(SimulationError::InvalidSubcktInstance(line.to_string())); }
            Ok(Some(SpiceDevice {
                name: parts[0].to_string(),
                kind: SpiceDeviceKind::Jj,
                connections: vec![parts[1].to_string(), parts[2].to_string()],
                params: BTreeMap::new(),
            }))
        }
        'R' | 'r' | 'L' | 'l' | 'C' | 'c' => {
            let kind = match first_char {
                'R' | 'r' => SpiceDeviceKind::Resistor,
                'L' | 'l' => SpiceDeviceKind::Inductor,
                'C' | 'c' => SpiceDeviceKind::Capacitor,
                _ => unreachable!(),
            };
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() < 4 { return Err(SimulationError::InvalidSubcktInstance(line.to_string())); }
            Ok(Some(SpiceDevice {
                name: parts[0].to_string(),
                kind,
                connections: vec![parts[1].to_string(), parts[2].to_string()],
                params: BTreeMap::new(),
            }))
        }
        _ => Ok(None),
    }
}
```

```rust
pub fn spice_to_ir(devices: &[SpiceDevice]) -> Result<rflux_ir::Netlist, String> {
    let mut netlist = rflux_ir::Netlist::new();
    let mut node_map: HashMap<String, rflux_ir::NodeId> = HashMap::new();

    for device in devices {
        match &device.kind {
            SpiceDeviceKind::SubcktInstance(subckt_name) => {
                let logic_op = match subckt_name.as_str() {
                    "and" => Some(rflux_ir::LogicOp::And),
                    "or" => Some(rflux_ir::LogicOp::Or),
                    "not" => Some(rflux_ir::LogicOp::Not),
                    "xor" => Some(rflux_ir::LogicOp::Xor),
                    "dff" => Some(rflux_ir::LogicOp::DffEnable),
                    _ => None,
                };
                let kind = if subckt_name == "dff" {
                    rflux_ir::NodeKind::Dff
                } else {
                    rflux_ir::NodeKind::CellInstance
                };
                let node = netlist.add_node_with_logic(kind, device.name.clone(), logic_op);
                node_map.insert(device.name.clone(), node);
            }
            SpiceDeviceKind::Jj => {
                let node = netlist.add_node(rflux_ir::NodeKind::CellInstance, device.name.clone());
                node_map.insert(device.name.clone(), node);
            }
            _ => {}
        }
    }

    Ok(netlist)
}
```

```rust
pub fn map_netlist_area_optimized(&self, netlist: &'a Netlist) -> Vec<TechMappedNode<'a>> {
    // For each node, find all matching cells and choose the smallest
    netlist.nodes().iter().filter_map(|node| {
        let kind = Self::map_kind(&node.kind);
        let best_cell = self.pdk.cell_library.cells.iter()
            .filter(|c| c.kind == kind)
            .min_by(|a, b| a.area_um2.partial_cmp(&b.area_um2).unwrap_or(std::cmp::Ordering::Equal));
        best_cell.map(|cell| TechMappedNode { node, cell })
    }).collect()
}
```
